00030 // The version is stored as 4 bytes after the signature and also serves as a
00031 // byte order mark. Signature and version combined are 16 bytes long.

00044 bool DepsLog::OpenForWrite(const string& path, string* err) {
00045   if (needs_recompaction_) {
00046     if (!Recompact(path, err))
00047       return false;
00048   }
00049   
00050   file_ = fopen(path.c_str(), "ab");
00051   if (!file_) {
00052     *err = strerror(errno);
00053     return false;
00054   }
00055   setvbuf(file_, NULL, _IOFBF, kMaxBufferSize);
00056   SetCloseOnExec(fileno(file_));
00057 
00058   // Opening a file in append mode doesn't set the file pointer to the file's
00059   // end on Windows. Do that explicitly.
00060   fseek(file_, 0, SEEK_END);
00061 
00062   if (ftell(file_) == 0) {
00063     if (fwrite(kFileSignature, sizeof(kFileSignature) - 1, 1, file_) < 1) {
00064       *err = strerror(errno);
00065       return false;
00066     }
00067     if (fwrite(&kCurrentVersion, 4, 1, file_) < 1) {
00068       *err = strerror(errno);
00069       return false;
00070     }
00071   }
00072   fflush(file_);
00073 
00074   return true;

00083 bool DepsLog::RecordDeps(Node* node, TimeStamp mtime,
00084                          int node_count, Node** nodes) {
00085   // Track whether there's any new data to be recorded.
00086   bool made_change = false;
00087 
00088   // Assign ids to all nodes that are missing one.
00089   if (node->id() < 0) {
00090     RecordId(node);
00091     made_change = true;
00092   }
00093   for (int i = 0; i < node_count; ++i) {
00094     if (nodes[i]->id() < 0) {
00095       RecordId(nodes[i]);
00096       made_change = true;
00097     }
00098   }
00099 
00100   // See if the new data is different than the existing data, if any.
00101   if (!made_change) {
00102     Deps* deps = GetDeps(node);
00103     if (!deps ||
00104         deps->mtime != mtime ||
00105         deps->node_count != node_count) {
00106       made_change = true;
00107     } else {
00108       for (int i = 0; i < node_count; ++i) {
00109         if (deps->nodes[i] != nodes[i]) {
00110           made_change = true;
00111           break;
00112         }
00113       }
00114     }
00115   }
00116 
00117   // Don't write anything if there's no new info.
00118   if (!made_change)
00119     return true;
00120 
00121   // Update on-disk representation.
00122   uint16_t size = 4 * (1 + 1 + (uint16_t)node_count);
00123   size |= 0x8000;  // Deps record: set high bit.
00124   fwrite(&size, 2, 1, file_);
00125   int id = node->id();
00126   fwrite(&id, 4, 1, file_);
00127   int timestamp = mtime;
00128   fwrite(&timestamp, 4, 1, file_);
00129   for (int i = 0; i < node_count; ++i) {
00130     id = nodes[i]->id();
00131     fwrite(&id, 4, 1, file_);
00132   }
00133   fflush(file_);
00134 
00135   // Update in-memory representation.
00136   Deps* deps = new Deps(mtime, node_count);
00137   for (int i = 0; i < node_count; ++i)
00138     deps->nodes[i] = nodes[i];
00139   UpdateDeps(node->id(), deps);
00140 
00141   return true;
00142 }

00151   METRIC_RECORD(".ninja_deps load");

00153   FILE* f = fopen(path.c_str(), "rb");
00154   if (!f) {
00155     if (errno == ENOENT)
00156       return true;
00157     *err = strerror(errno);
00158     return false;
00159   }
00160 
00161   bool valid_header = true;

00163   if (!fgets(buf, sizeof(buf), f) || fread(&version, 4, 1, f) < 1)
00164     valid_header = false;
00165   if (!valid_header || strcmp(buf, kFileSignature) != 0 ||
00166       version != kCurrentVersion) {
00167     *err = "bad deps log signature or version; starting over";
00168     fclose(f);
00169     unlink(path.c_str());
00170     // Don't report this as a failure.  An empty deps log will cause
00171     // us to rebuild the outputs anyway.
00172     return true;
00173   }
00174 
00175   long offset;
00176   bool read_failed = false;
00177   int unique_dep_record_count = 0;
00178   int total_dep_record_count = 0;
00179   for (;;) {
00180     offset = ftell(f);
00181 
00182     uint16_t size;
00183     if (fread(&size, 2, 1, f) < 1) {
00184       if (!feof(f))
00185         read_failed = true;
00186       break;
00187     }
00188     bool is_deps = (size >> 15) != 0;
00189     size = size & 0x7FFF;
00190 
00191     if (fread(buf, size, 1, f) < 1) {
00192       read_failed = true;
00193       break;
00194     }
00195 
00196     if (is_deps) {
00197       assert(size % 4 == 0);
00198       int* deps_data = reinterpret_cast<int*>(buf);
00199       int out_id = deps_data[0];
00200       int mtime = deps_data[1];
00201       deps_data += 2;
00202       int deps_count = (size / 4) - 2;
00203 
00204       Deps* deps = new Deps(mtime, deps_count);
00205       for (int i = 0; i < deps_count; ++i) {
00206         assert(deps_data[i] < (int)nodes_.size());
00207         assert(nodes_[deps_data[i]]);
00208         deps->nodes[i] = nodes_[deps_data[i]];
00209       }
00210 
00211       total_dep_record_count++;
00212       if (!UpdateDeps(out_id, deps))
00213         ++unique_dep_record_count;
00214     } else {
00215       StringPiece path(buf, size);
00216       Node* node = state->GetNode(path);
00217       assert(node->id() < 0);
00218       node->set_id(nodes_.size());
00219       nodes_.push_back(node);
00220     }
00221   }
00222 
00223   if (read_failed) {
00224     // An error occurred while loading; try to recover by truncating the
00225     // file to the last fully-read record.
00226     if (ferror(f)) {
00227       *err = strerror(ferror(f));
00228     } else {
00229       *err = "premature end of file";
00230     }
00231     fclose(f);
00232 
00233     if (!Truncate(path.c_str(), offset, err))
00234       return false;
00235 
00236     // The truncate succeeded; we'll just report the load error as a
00237     // warning because the build can proceed.
00238     *err += "; recovering";
00239     return true;
00240   }
00241 
00242   fclose(f);
00243 
00244   // Rebuild the log if there are too many dead records.
00245   int kMinCompactionEntryCount = 1000;
00246   int kCompactionRatio = 3;
00247   if (total_dep_record_count > kMinCompactionEntryCount &&
00248       total_dep_record_count > unique_dep_record_count * kCompactionRatio) {
00249     needs_recompaction_ = true;
00250   }
00251

00263 bool DepsLog::Recompact(const string& path, string* err) {
00264   METRIC_RECORD(".ninja_deps recompact");
00265   printf("Recompacting deps...\n");
00266 
00267   Close();
00268   string temp_path = path + ".recompact";
00269 
00270   // OpenForWrite() opens for append.  Make sure it's not appending to a
00271   // left-over file from a previous recompaction attempt that crashed somehow.
00272   unlink(temp_path.c_str());
00273 
00274   DepsLog new_log;
00275   if (!new_log.OpenForWrite(temp_path, err))
00276     return false;
00277 
00278   // Clear all known ids so that new ones can be reassigned.  The new indices
00279   // will refer to the ordering in new_log, not in the current log.
00280   for (vector<Node*>::iterator i = nodes_.begin(); i != nodes_.end(); ++i)
00281     (*i)->set_id(-1);
00282   
00283   // Write out all deps again.
00284   for (int old_id = 0; old_id < (int)deps_.size(); ++old_id) {
00285     Deps* deps = deps_[old_id];
00286     if (!deps) continue;  // If nodes_[old_id] is a leaf, it has no deps.
00287 
00288     if (!new_log.RecordDeps(nodes_[old_id], deps->mtime,
00289                             deps->node_count, deps->nodes)) {
00290       new_log.Close();
00291       return false;
00292     }
00293   }
00294 
00295   new_log.Close();
00296 
00297   // All nodes now have ids that refer to new_log, so steal its data.
00298   deps_.swap(new_log.deps_);
00299   nodes_.swap(new_log.nodes_);
00300 
00301   if (unlink(path.c_str()) < 0) {
00302     *err = strerror(errno);
00303     return false;
00304   }
00305 
00306   if (rename(temp_path.c_str(), path.c_str()) < 0) {
00307     *err = strerror(errno);
00308     return false;
00309   }
00310 
00311   return true;
00312 }
