00020 /// Interface for accessing the disk.
00021 ///
00022 /// Abstract so it can be mocked out for tests.  The real implementation
00023 /// is RealDiskInterface.
00024 struct DiskInterface {
00025   virtual ~DiskInterface() {}
00026 
00027   /// stat() a file, returning the mtime, or 0 if missing and -1 on
00028   /// other errors.
00029   virtual int Stat(const std::string& path) = 0;
00030 
00031   /// Create a directory, returning false on failure.
00032   virtual bool MakeDir(const std::string& path) = 0;
00033 
00034   /// Read a file to a string.  Fill in |err| on error.
00035   virtual std::string ReadFile(const std::string& path, std::string* err) = 0;
00036 
00037   /// Remove the file named @a path. It behaves like 'rm -f path' so no errors
00038   /// are reported if it does not exists.
00039   /// @returns 0 if the file has been removed,
00040   ///          1 if the file does not exist, and
00041   ///          -1 if an error occurs.
00042   virtual int RemoveFile(const std::string& path) = 0;
00043 
00044   /// Create all the parent directories for path; like mkdir -p
00045   /// `basename path`.
00046   bool MakeDirs(const std::string& path);
00047 };
