00040 bool Subprocess::Start(SubprocessSet* set, const string& command) {
00041   int output_pipe[2];
00042   if (pipe(output_pipe) < 0)
00043     Fatal("pipe: %s", strerror(errno));
00044   fd_ = output_pipe[0];
00045   SetCloseOnExec(fd_);
00046 
00047   pid_ = fork();
00048   if (pid_ < 0)
00049     Fatal("fork: %s", strerror(errno));
00050 
00051   if (pid_ == 0) {
00052     close(output_pipe[0]);
00053 
00054     // Track which fd we use to report errors on.
00055     int error_pipe = output_pipe[1];
00056     do {
00057       // Open /dev/null over stdin.
00058       int devnull = open("/dev/null", O_WRONLY);
00059       if (devnull < 0)
00060         break;
00061       if (dup2(devnull, 0) < 0)
00062         break;
00063       close(devnull);
00064 
00065       if (dup2(output_pipe[1], 1) < 0 ||
00066           dup2(output_pipe[1], 2) < 0)
00067         break;
00068 
00069       // Now can use stderr for errors.
00070       error_pipe = 2;
00071       close(output_pipe[1]);
00072 
00073       execl("/bin/sh", "/bin/sh", "-c", command.c_str(), NULL);
00074     } while (false);
00075 
00076     // If we get here, something went wrong; the execl should have
00077     // replaced us.
00078     char* err = strerror(errno);
00079     if (write(error_pipe, err, strlen(err)) < 0) {
00080       // If the write fails, there's nothing we can do.
00081       // But this block seems necessary to silence the warning.
00082     }
00083     _exit(1);
00084   }
00085 
00086   close(output_pipe[1]);
00087   return true;
00088 }
